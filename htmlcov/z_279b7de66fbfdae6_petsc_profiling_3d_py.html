1""" 

23D PETSc profiling for the DPP model using manufactured 3D solutions (paper Eq. 6.3). 

3 

4This mirrors perphil.experiments.petsc_profiling but swaps the mesh to 3D and 

5uses exact_expressions_3d for Dirichlet boundary conditions.

43def run_perf_once_3d( 

44 nx: int, 

45 approach: Approach, 

46 eager: bool = True, 

47 logical_events: Optional[List[str]] = None, 

48 repeats: int = 5, 

49 backend: str = "auto", 

50) -> Dict[str, Any]: 

51 """ 

52 3D analog of prof2d.run_perf_once, using UnitCube meshes and 3D MMS BCs. 

53 Returns a flattened dict ready for DataFrame assembly. 

54 """ 

55 prof2d.ensure_petsc_logging() 

56 

57 mesh = _build_mesh_3d(nx) 

58 W = _build_spaces_3d(mesh) 

59 comm = mesh.comm 

60 

61 params = _default_model_params() 

62 

63 # Manufactured Dirichlet BCs on boundary from 3D MMS 

64 _u1e, p1e, _u2e, p2e = exact_expressions_3d(mesh, params) 

65 bcs = [ 

66 fd.DirichletBC(W.sub(0), p1e, "on_boundary"), 

67 fd.DirichletBC(W.sub(1), p2e, "on_boundary"), 

68 ] 

69 

70 logical_events = list(dict.fromkeys((logical_events or []) + prof2d.DEFAULT_LOGICAL_EVENTS)) 

71 

72 # Warmup 

73 if eager: 

74 if approach == Approach.PICARD_MUMPS: 

75 sol_warm = solve_dpp_nonlinear( 

76 W, params, bcs=bcs, solver_parameters={**params_for(approach)} 

77 ) 

78 else: 

79 sol_warm = solve_dpp(W, params, bcs=bcs, solver_parameters={**params_for(approach)}) 

80 prof2d._enable_convergence_history_if_possible(prof2d._extract_solution_handle(sol_warm)) 

81 

82 def run_once() -> None: 

83 if approach == Approach.PICARD_MUMPS: 

84 solve_dpp_nonlinear(W, params, bcs=bcs, solver_parameters={**params_for(approach)}) 

85 else: 

86 solve_dpp(W, params, bcs=bcs, solver_parameters={**params_for(approach)}) 

87 

88 # Record RSS before profiling this case (per-rank) 

89 rss_before_kb = prof2d._get_rss_kb() 

90 

91 # Select backend(s) 

92 backends = [backend] if backend != "auto" else ["json", "ascii", "events", "stage", "wall"] 

93 

94 wall_total = 0.0 

95 for backend_name in backends: 

96 try: 

97 if backend_name == "json": 

98 event_times, event_flops, wall_time = prof2d._profile_with_log_view_json( 

99 comm, run_once, logical_events, repeats 

100 ) 

101 elif backend_name == "ascii": 

102 event_times, event_flops, wall_time = prof2d._profile_with_log_view_ascii( 

103 comm, run_once, logical_events, repeats 

104 ) 

105 elif backend_name == "events": 

106 event_times, event_flops, wall_time = prof2d._profile_with_events_api( 

107 comm, run_once, logical_events, repeats 

108 ) 

109 elif backend_name == "stage": 

110 event_times, event_flops, wall_time = prof2d._profile_with_stage_api( 

111 comm, run_once, logical_events, repeats 

112 ) 

113 else: 

114 # wall fallback 

115 import time as _t 

116 

117 comm.barrier() 

118 t0 = _t.perf_counter() 

119 for _ in range(max(1, repeats)): 

120 run_once() 

121 wall_time = _t.perf_counter() - t0 

122 event_times = {e: 0.0 for e in logical_events} 

123 event_times["KSPSolve"] = wall_time 

124 event_flops = {e: 0.0 for e in logical_events} 

125 if (sum(event_times.values()) > 0.0) or backend_name in ("wall", "events"): 

126 backend_used = backend_name 

127 wall_total = wall_time 

128 break 

129 except Exception: 

130 continue 

131 else: 

132 # If all backends failed 

133 import time as _t 

134 

135 comm.barrier() 

136 t0 = _t.perf_counter() 

137 for _ in range(max(1, repeats)): 

138 run_once() 

139 wall_time = _t.perf_counter() - t0 

140 event_times = {e: 0.0 for e in logical_events} 

141 event_times["KSPSolve"] = wall_time 

142 event_flops = {e: 0.0 for e in logical_events} 

143 backend_used = "wall" 

144 wall_total = wall_time 

145 

146 # Final solve to collect iterations/residual and build operators for memory stats 

147 if approach == Approach.PICARD_MUMPS: 

148 sol = solve_dpp_nonlinear(W, params, bcs=bcs, solver_parameters={**params_for(approach)}) 

149 else: 

150 sol = solve_dpp(W, params, bcs=bcs, solver_parameters={**params_for(approach)}) 

151 

152 # Metadata and counters 

153 num_cells = comm.allreduce(mesh.cell_set.size, op=prof2d.MPI.SUM) 

154 dofs = W.dim() 

155 solver_handle = prof2d._extract_solution_handle(sol) 

156 ksp_iters = prof2d._extract_ksp_iters_if_possible(solver_handle) 

157 iterations = ksp_iters if ksp_iters is not None else getattr(sol, "iteration_number", None) 

158 residual = float(getattr(sol, "residual_error", float("nan"))) 

159 

160 # Memory 

161 rss_after_kb = prof2d._get_rss_kb() 

162 rss_peak_kb = float(comm.allreduce(rss_after_kb, op=prof2d.MPI.MAX)) 

163 rss_delta_kb = float(comm.allreduce(max(0, rss_after_kb - rss_before_kb), op=prof2d.MPI.MAX)) 

164 mat_mem = prof2d._collect_matrix_memory(solver_handle) 

165 memory: Dict[str, float] = {"rss_peak_kb": rss_peak_kb, "rss_delta_kb": rss_delta_kb, **mat_mem} 

166 

167 meta: Dict[str, Any] = { 

168 "petsc_version": PETSc.Sys.getVersion(), 

169 "firedrake_version": getattr(fd, "__version__", None), 

170 "backend": backend_used, 

171 "repeats": repeats, 

172 } 

173 

174 # Flatten like prof2d.PerfResult.to_dict() 

175 row: Dict[str, Any] = { 

176 "approach": approach.value, 

177 "nx": nx, 

178 "ny": nx, 

179 "dofs": int(dofs), 

180 "num_cells": int(num_cells), 

181 "iterations": int(iterations) if iterations is not None else None, 

182 "residual": residual, 

183 "time_total": float(wall_total / max(1, repeats)), 

184 "time_total_repeats": float(wall_total), 

185 } 

186 # times/flops 

187 for k, v in (event_times or {}).items(): 

188 row[f"time_{k}"] = float(v) 

189 for k, v in (event_flops or {}).items(): 

190 row[f"flops_{k}"] = float(v) 

191 t = float(event_times.get(k, 0.0)) 

192 row[f"mflops_{k}"] = (float(v) / t / 1e6) if t > 0.0 else 0.0 

193 row["flops_total"] = float(sum((event_flops or {}).values())) 

194 # memory 

195 for k, v in memory.items(): 

196 row[f"mem_{k}"] = float(v) 

197 # metadata (optional): we can add to row if desired or keep separate 

198 row["backend"] = meta.get("backend") 

199 row["repeats"] = meta.get("repeats") 

200 return row

203def run_perf_sweep_3d( 

204 mesh_sizes: List[int], 

205 approaches: List[Approach], 

206 logical_events: Optional[List[str]] = None, 

207 eager: bool = True, 

208 repeats: int = 5, 

209 backend: str = "auto", 

210) -> pd.DataFrame: 

211 rows: List[Dict[str, Any]] = [] 

212 for nx in mesh_sizes: 

213 for ap in approaches: 

214 res = run_perf_once_3d( 

215 nx, 

216 ap, 

217 eager=eager, 

218 logical_events=logical_events, 

219 repeats=repeats, 

220 backend=backend, 

221 ) 

222 rows.append(res) 

223 PETSc.Sys.Print( 

224 f"[perf3d] nx={nx} {ap.value}: iters={res.get('iterations')}, " 

225 f"time_total={res.get('time_total', 0.0):.3e}s, " 

226 f"KSPSolve={res.get('time_KSPSolve', 0.0):.3e}s, " 

227 f"flops_total={res.get('flops_total', 0.0):.3e} " 

228 f"(backend={res.get('backend')}, repeats={res.get('repeats')})" 

229 ) 

230 return pd.DataFrame(rows)
